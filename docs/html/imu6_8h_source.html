203 static inline void caerIMU6EventValidate(caerIMU6Event event, caerIMU6EventPacket packet) {
204  if (!caerIMU6EventIsValid(event)) {
205  SET_NUMBITS32(event->info, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);
206 
207  // Also increase number of events and valid events.
208  // Only call this on (still) invalid events!
209  caerEventPacketHeaderSetEventNumber(&packet->packetHeader,
210  caerEventPacketHeaderGetEventNumber(&packet->packetHeader) + 1);
211  caerEventPacketHeaderSetEventValid(&packet->packetHeader,
212  caerEventPacketHeaderGetEventValid(&packet->packetHeader) + 1);
213  }
214  else {
215  caerLog(CAER_LOG_CRITICAL, "IMU6 Event", "Called caerIMU6EventValidate() on already valid event.");
216  }

228 static inline void caerIMU6EventInvalidate(caerIMU6Event event, caerIMU6EventPacket packet) {
229  if (caerIMU6EventIsValid(event)) {
230  CLEAR_NUMBITS32(event->info, VALID_MARK_SHIFT, VALID_MARK_MASK);
