501 static inline bool caerEventPacketEquals(caerEventPacketHeaderConst firstPacket,
502  caerEventPacketHeaderConst secondPacket) {
503  // If any of the packets is NULL, they can't be equal.
504  if (firstPacket == NULL || secondPacket == NULL) {
505  return (false);
506  }
507 
508  // If both packets are the same packet (pointer equal),
509  // they are of course indeed equal packets.
510  if (firstPacket == secondPacket) {
511  return (true);
512  }
513 
514  // Actually compare memory now. We compare header equality, and
515  // all events up to eventNumber. The remaining events up to
516  // eventCapacity are by definition all zeroed out, so must be
517  // equal, if the capacity is the same, which it is, as we check
518  // for that when ensuring header equality.
519  if (memcmp(firstPacket, secondPacket, CAER_EVENT_PACKET_HEADER_SIZE) != 0) {
520  return (false);
521  }
522 
523  size_t memCmpSize = (size_t) (caerEventPacketHeaderGetEventNumber(firstPacket)
524  * caerEventPacketHeaderGetEventSize(firstPacket));
525  if (memcmp(((const uint8_t *) firstPacket) + CAER_EVENT_PACKET_HEADER_SIZE,
526  ((const uint8_t *) secondPacket) + CAER_EVENT_PACKET_HEADER_SIZE, memCmpSize) != 0) {
527  return (false);
528  }
529 
530  return (true);
531 }

563 static inline void caerEventPacketClean(caerEventPacketHeader packet) {
564  // Handle empty event packets.
565  if (packet == NULL) {
566  return;
567  }
568 
569  // Calculate needed memory for new event packet.
570  int32_t eventValid = caerEventPacketHeaderGetEventValid(packet);
571  int32_t eventNumber = caerEventPacketHeaderGetEventNumber(packet);
572 
573  // If we have no invalid events, we're already done.
574  if (eventValid == eventNumber) {
575  return;
576  }
577 
578  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet);
579  int32_t eventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
580 
581  // Move all valid events close together. Must check every event for validity!
582  size_t offset = CAER_EVENT_PACKET_HEADER_SIZE;
583 
584  CAER_ITERATOR_VALID_START(packet, const void *)
585  void *dest = ((uint8_t *) packet) + offset;
586 
587  if (dest != caerIteratorElement) {
588  memcpy(dest, caerIteratorElement, (size_t) eventSize);
589  offset += (size_t) eventSize;
590  }
591  }
592 
593  // Reset remaining memory, up to capacity, to zero (all events invalid).
594  memset(((uint8_t *) packet) + offset, 0, (size_t) ((eventCapacity - eventValid) * eventSize));
595 
596  // Event capacity remains unchanged, event number shrunk to event valid number.
597  caerEventPacketHeaderSetEventNumber(packet, eventValid);
598 }

618 static inline caerEventPacketHeader caerEventPacketResize(caerEventPacketHeader packet, int32_t newEventCapacity) {
619  if (packet == NULL || newEventCapacity <= 0) {
620  return (NULL);
621  }
622 
623  // Always clean for consistency with shrink case (side-effects guarantee).
624  caerEventPacketClean(packet);
625 
626  int32_t oldEventCapacity = caerEventPacketHeaderGetEventCapacity(packet);
627 
628  if (oldEventCapacity == newEventCapacity) {
629  // Nothing to do in this case.
630  return (packet);
631  }
632 
633  int32_t eventSize = caerEventPacketHeaderGetEventSize(packet);
634  size_t newEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (newEventCapacity * eventSize);
635 
636  // Reallocate memory used to hold events.
637  packet = (caerEventPacketHeader) realloc(packet, newEventPacketSize);
638  if (packet == NULL) {
639  caerLog(CAER_LOG_CRITICAL, "Event Packet",
640  "Failed to reallocate %zu bytes of memory for resizing Event Packet of capacity %"
641  PRIi32 " to new capacity of %" PRIi32 ". Error: %d.", newEventPacketSize, oldEventCapacity,
642  newEventCapacity, errno);
643  return (NULL);
644  }
645 
646  if (newEventCapacity > oldEventCapacity) {
647  // Capacity increased: we simply zero out the newly added events.
648  size_t oldEventPacketSize = CAER_EVENT_PACKET_HEADER_SIZE + (size_t) (oldEventCapacity * eventSize);
649 
650  memset(((uint8_t *) packet) + oldEventPacketSize, 0,
651  (size_t) ((newEventCapacity - oldEventCapacity) * eventSize));
652  }
653  else {
654  // Capacity decreased: the events were cleaned, so eventValid == eventNumber.
655  // They also are all together in memory. Thus we can simply keep the current
656  // eventValid/eventNumber counts if the capacity is still bigger or equal to
657  // them, or, if new capacity is smaller, we reset them to that value.
658  int32_t oldEventNumber = caerEventPacketHeaderGetEventNumber(packet);
659 
660  if (newEventCapacity < oldEventNumber) {
661  caerEventPacketHeaderSetEventValid(packet, newEventCapacity);
662  caerEventPacketHeaderSetEventNumber(packet, newEventCapacity);
663  }
664  }
665 
666  // Update capacity header field.
667  caerEventPacketHeaderSetEventCapacity(packet, newEventCapacity);
668 
669  return (packet);
670 }
