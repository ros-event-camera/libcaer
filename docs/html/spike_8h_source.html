201 static inline void caerSpikeEventValidate(caerSpikeEvent event, caerSpikeEventPacket packet) {
202  if (!caerSpikeEventIsValid(event)) {
203  SET_NUMBITS32(event->data, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);
204 
205  // Also increase number of events and valid events.
206  // Only call this on (still) invalid events!
207  caerEventPacketHeaderSetEventNumber(&packet->packetHeader,
208  caerEventPacketHeaderGetEventNumber(&packet->packetHeader) + 1);
209  caerEventPacketHeaderSetEventValid(&packet->packetHeader,
210  caerEventPacketHeaderGetEventValid(&packet->packetHeader) + 1);
211  }
212  else {
213  caerLog(CAER_LOG_CRITICAL, "Spike Event", "Called caerSpikeEventValidate() on already valid event.");
214  }
215 }

314 static inline uint16_t caerSpikeEventGetY(caerSpikeEventConst event) {
315  uint8_t chipId = caerSpikeEventGetChipID(event);
316  uint8_t coreId = caerSpikeEventGetSourceCoreID(event);
317  uint32_t neuronId = caerSpikeEventGetNeuronID(event);
318 
319  uint16_t columnId = (neuronId & 0x0F);
320  bool addColumn = (coreId & 0x01);
321  bool addColumnChip = (chipId & (0x01 << 2));
322  columnId = U16T(columnId + (addColumn) * 16 + (addColumnChip) * 32);
323 
324  return (columnId);
325 }
