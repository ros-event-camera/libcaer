202 static inline void caerEarEventValidate(caerEarEvent event, caerEarEventPacket packet) {
203  if (!caerEarEventIsValid(event)) {
204  SET_NUMBITS32(event->data, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);
205 
206  // Also increase number of events and valid events.

209  caerEventPacketHeaderGetEventNumber(&packet->packetHeader) + 1);
