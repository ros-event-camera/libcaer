202 static inline void caerPolarityEventValidate(caerPolarityEvent event, caerPolarityEventPacket packet) {
203  if (!caerPolarityEventIsValid(event)) {
204  SET_NUMBITS32(event->data, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);
205 
206  // Also increase number of events and valid events.
