198 static inline void caerSampleEventValidate(caerSampleEvent event, caerSampleEventPacket packet) {
199  if (!caerSampleEventIsValid(event)) {
200  SET_NUMBITS32(event->data, VALID_MARK_SHIFT, VALID_MARK_MASK, 1);
201 
202  // Also increase number of events and valid events.
203  // Only call this on (still) invalid events!
204  caerEventPacketHeaderSetEventNumber(&packet->packetHeader,
205  caerEventPacketHeaderGetEventNumber(&packet->packetHeader) + 1);
206  caerEventPacketHeaderSetEventValid(&packet->packetHeader,
207  caerEventPacketHeaderGetEventValid(&packet->packetHeader) + 1);
208  }
209  else {
